Use `console.dir()` to print nested object.

```js
const obj = {
	a: 'a',
	b: {
		c: 'c',
		d: {
			e: 'e',
			f: {
				g: 'g',
				h: {
					i: 'i'
				}
			}
		}
	}
};

console.log(obj);
// { a: 'a', b: { c: 'c', d: { e: 'e', f: [Object] } } }

console.dir(obj, { depth: null }); // `depth: null` ensures unlimited recursion
// {
//   a: 'a',
//   b: {
//     c: 'c',
//     d: { e: 'e', f: { g: 'g', h: { i: 'i' } } }
//   }
// }
```
